// Creation d'un "world" de 15x15 cases
World world = new World(new BlockSurface(15, 15),"world-test");

// Initialisation du constructeur, editeur de "world"
WorldBuilder wb = new WorldBuilder(world);

// On ajoute une surface de 15x15 , a partir de la position (0,0). Toutes les cases
// Sur cette surface seront de type BlockType.dev_nice
wb.addPatternBlock(15,15,BlockPosition.origin,BlockType.dev_nice,true);

// On ajoute une surface de 5x5 , a partir de la position (3,2), de type
// BlockType.dev_notPassable, FALSE indique qu'il ne sera pas possible
// de passer sur ces cases
wb.addPatternBlock(5,5,new BlockPosition(3, 2),BlockType.dev_notPassable,false);

// On ajoute la surface au "world"
wb.buildWorld();
