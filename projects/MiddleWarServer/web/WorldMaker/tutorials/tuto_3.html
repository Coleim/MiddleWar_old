TUTO - 3 - Un peu de design

// Creation d'un "world" de 15x15 cases
World world = new World(new BlockSurface(15, 15),"world-test");

// Initialisation du constructeur, editeur de "world"
WorldBuilder wb = new WorldBuilder(world);

// 15x15, position (0,0), BlockType.grass_C
wb.addPatternBlock(15,15,BlockPosition.origin,BlockType.grass_C,true);

// 4x4, position (3,2), SurfaceType.town_ground_1_grass
// La fonction addPatternSquare permet de gérer automatiquement un rectangle
// d'une surface donnée (angles, ect...)
wb.addPatternSquare(4,4,new BlockPosition(3, 2),SurfaceType.town_ground_1_grass,false);

// On ajoute la surface au "world"
wb.buildWorld();
